1.	Lister tous les livres écrits par l’auteur Amadou Hampaté Bâ.

	FROM livre l

2.	Afficher les étudiants inscrits en L3 Informatique.
SQL : 

	SELECT e.nomEtudiant,e.prenomEtudiant,e.niveau,e.filiere
	FROM etudiant e
	WHERE e.filiere = 'Informatique' 
		AND e.niveau = 'L3'

3. Trouver les livres disponibles (c’est-à-dire sans emprunt en cours).
	SELECT * 
	FROM livre l
	WHERE l.idLivre NOT IN
		(SELECT e.fk_idLivre
		FROM emprunt e 
		WHERE e.dateRetour IS NULL);

5. Afficher les étudiants ayant emprunté un livre intitulé Introduction à la Base de Donnée.

# Distinct sert éliminer les doublons 
SELECT distinct E.*
FROM etudiant E
	Inner Join emprunt EM 
		ON E.idEtudiant = EM.fk_idEtudiant
	Inner Join livre L 
		ON EM.fk_idLivre = L.idLivre
WHERE L.titre = 'Introduction à la Base de Donnée';


6. Lister les livres empruntés entre le 1er janvier 2024 et le 31 mars 2024.
SELECT distinct L.titre 
from livre L
inner join emprunt EM ON L.idLivre = EM.id_livre
WHERE EM.date_emprunt between '2024-01-01' AND '2024-03-31'

9. Afficher les étudiants ayant au moins deux emprunts distincts.
SELECT idEtudiant,nomEtudiant,prenomEtudiant,COUNT(distinct EM.idemprunt) NombreEmprunt
from etudiant E
join emprunt EM
ON E.idEtudiant =EM.fk_idEtudiant
Group by idEtudiant,nomEtudiant,prenomEtudiant
HAVING NombreEmprunt >= 2
	
10. Trouver le ou les éditeurs qui ont publié le plus de livres.

SELECT ED.idEditeur,ED.nomEditeur,COUNT(distinct L.idlivre) Nombrepublication
from Editeur ED
join livre L
ON ED.idEditeur = L.fk_idEditeur
Group by ED.idEditeur,ED.nomEditeur 
HAVING Nombrepublication = (SELECT MAX(Nombrepublication1)
					from ( SELECT L1.fk_idEditeur, COUNT(distinct L1.idLivre) Nombrepublication1
							FROM livre L1
							GROUP BY fk_idEditeur
						) 	AS sous_requete
			)
